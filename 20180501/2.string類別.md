### 字串的附加
```
#include <iostream>
using namespace std;

int main()
{
  string s1("Welcome");
  s1.append(" to CPP"); 
  cout << s1 << endl; 
}
```
```
string s1("Welcome");
s1.append(" to CPP"); 
cout << s1 << endl; 
```
```
string s2("Welcome");
s2.append(" to C and Cpp", 3, 6); 
cout << s2 << endl;
```
```
string s3("Welcome");
s3.append(" to C and Cpp", 7); 
cout << s3 << endl; 
```
```
string s4("Welcome"); 
s4.append(14, 'F'); 
cout << s4 << endl; 
```
### 從螢幕讀取字串
```
string city;
cout << "Enter a city: ";
cin >> city; // Read to array city
cout << "You entered " << city << endl;

```

```
string city;
cout << "Enter a city: ";
getline(cin, city, '\n'); // Same as getline(cin, city)
cout << "You entered " << city << endl;
```
